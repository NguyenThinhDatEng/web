2. attributes
   class
   "{class_name}"

   height
   "{height of image}px"

   hidden

   id
   "{id}" --> id have to begin with the letter

   style
   "color: ...,
   font-size: ...px
   font-weight: ...,
   border: {number}px solid {color}"

   target
   "\_blank" -> open on other tab
   "\_parent" -> open on current tab
   href
   "{link}"

   width
   "{width of image}"
   src
   "{path of image}"
   action
   "{link server}"
   method
   "POST"
   "GET"
